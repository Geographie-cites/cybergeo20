\jframe{Extraction des mots-clés brute}{

\textit{Text-mining en python avec nltk}

\begin{itemize}
\item Detection de la langue par \emph{stop-words} % TODO language stats
\item Parsing/tokenizing/pos-tagging/stemming effectué différemment selon la langue :
\begin{itemize}
\item Anglais : pos-tagger intégré à \texttt{nltk}, combiné à un \emph{PorterStemmer}
\item Français ou autre : utilisation de \texttt{TreeTagger}
\end{itemize}
\item Selection des \emph{n-grams} potentiels (avec $1 \leq n \leq 4$) : anglais $\bigcap \{NN \cup VBG \cup JJ \}$ ;  français  $\bigcap \{NOM \cup ADJ\}$
\item Insertion en base pour extraction quasi-instantanée plus tard (10j $\rightarrow$ 5min)
\end{itemize}

}


\jframe{Estimation de la pertinence par bootstrap}{
Estimation exacte de la pertinence en $\Theta (\sum_i N_i^2)$ : impossible sur un corpus où $\sum_i N_i^2 \simeq N <N_i>^2 \simeq 4\cdot 10^6$
}
